// Пример простого кода на языке 1С для тестирования грамматики

// Объявление переменных
Перем Сумма;
Перем Количество;
Перем ИтоговаяСумма;

// Процедура без параметров
Процедура ПосчитатьСумму()
    Сумма = 0;
    Для Счетчик = 1 По 10 Цикл
        Сумма = Сумма + Счетчик;
    КонецЦикла;
КонецПроцедуры

// Функция с параметрами и возвратом значения
Функция УмножитьНаКоэффициент(Значение, Коэффициент = 2) Экспорт
    Если Коэффициент <= 0 Тогда
        Возврат 0;
    ИначеЕсли Коэффициент = 1 Тогда
        Возврат Значение;
    Иначе
        Возврат Значение * Коэффициент;
    КонецЕсли;
КонецФункции

// Использование условий
Если Сумма > 100 Тогда
    ИтоговаяСумма = Сумма * 0.95; // Скидка 5%
ИначеЕсли Сумма > 50 Тогда
    ИтоговаяСумма = Сумма * 0.97; // Скидка 3%
Иначе
    ИтоговаяСумма = Сумма;
КонецЕсли;

// Использование циклов
Количество = 5;
Пока Количество > 0 Цикл
    Сообщить("Осталось: " + Количество);
    Количество = Количество - 1;
КонецЦикла;

// Обработка исключений
Попытка
    ИтоговаяСумма = УмножитьНаКоэффициент(Сумма, "некорректное_значение");
Исключение
    Сообщить("Ошибка при вычислении: " + ОписаниеОшибки());
    ИтоговаяСумма = Сумма;
КонецПопытки;
